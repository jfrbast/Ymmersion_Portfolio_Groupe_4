                package main

import (
	"fmt"
	"unicode"
)

func main() {
	motdepasse := ""
	fmt.Scanln(&motdepasse)
	fmt.Println(motdepassediff(motdepasse))

}

func motdepassediff(motdepasse string) string {
	isMAJ := false
	isCHAR := false
	difficulte := 0
	if len(motdepasse) >= 8 {
		difficulte += 1
	}
	for _, i := range motdepasse {
		if unicode.IsUpper(i) && isMAJ == false {
			difficulte += 1
			isMAJ = true

		}
		if i >= '!' && i <= '/' && isCHAR == false {
			difficulte += 1
			isCHAR = true
		}
	}

	if difficulte == 1 {
		return "moyen"

	}
	if difficulte == 2 {
		return "fort"
	}

	if difficulte >= 3 {
		return "tres fort"
	}

	return "Bof"

}
