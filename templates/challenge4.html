package main

import (
"bufio"
"fmt"
"os"
"strings"
)
                            
func analyzeText(text string) (int, int, string) {
text = strings.TrimSpace(text)
words := strings.Fields(text)
wordCount := len(words)
charCount := len(strings.ReplaceAll(text, " ", "")) 
longestWord := ""
for _, word := range words {
if len(word) > len(longestWord) {
longestWord = word
    }
}
                            
return wordCount, charCount, longestWord
}
