Challenge n°3 : Palindrome

Description of the test

Goal

Create a program that can detect whether a word entered by the user is a palindrome

Instructions

Write a program to check whether a word or phrase (without spaces) is a palindrome (ex: “radar” , “kayak”). Time limit 20 min

package main

import (
	"bufio"
	"fmt"
	"os"
	"strings"
)

// Fonction pour vérifier si une chaîne est un palindrome
func isPalindrome(s string) bool {
	// Convertir en minuscules et enlever les espaces
	s = strings.ToLower(strings.ReplaceAll(s, " ", ""))

	// Comparer les caractères en miroir
	n := len(s)
	for i := 0; i < n/2; i++ {
		if s[i] != s[n-1-i] {
			return false
		}
	}
	return true
}

func main() {
	scanner := bufio.NewScanner(os.Stdin)

	fmt.Println("🔄 Bienvenue dans le jeu du Palindrome !")
	fmt.Println("➡️  Entrez un mot ou une phrase (ou tapez 'exit' pour quitter) :")

	for {
		fmt.Print("> ")
		scanner.Scan()
		input := scanner.Text()

		// Vérifier si l'utilisateur veut quitter
		if strings.ToLower(input) == "exit" {
			fmt.Println("👋 Merci d'avoir joué ! À bientôt !")
			break
		}

		// Vérifier si le mot est un palindrome
		if isPalindrome(input) {
			fmt.Println("✅ C'est un palindrome ! 🎉")
		} else {
			fmt.Println("❌ Ce n'est pas un palindrome.")
		}
	}
}
